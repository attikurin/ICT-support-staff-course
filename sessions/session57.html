range()の使い方
# range(終了値)：0から終了値-1まで

    print(i)  # 0, 1, 2

for i in range(5, 8):
    print(i)  # 5, 6, 7

# range(開始値, 終了値, 増加値)
for i in range(0, 10, 2):
    print(i)  # 0, 2, 4, 6, 8

3.2 プログラミング的思考の実践

問題解決の4ステップ
  1. 分解（Decomposition） - 大きな問題を小さな部分問題に分ける
  2. パターン認識（Pattern Recognition） - 似たような処理や規則性を見つける
  3. 抽象化（Abstraction） - 重要な情報に絞り、不要な詳細を省く
  4. アルゴリズム設計（Algorithm Design） - 問題解決の手順を明確にする
実践例：クラス全体の成績処理

問題：30人分のテスト結果から、平均点・最高点・最低点・合格者数を求める

ステップ1：分解
  1. 30個の点数を入力または用意する
  2. 合計点を計算する
  3. 平均点を計算する
  4. 最高点を見つける
  5. 最低点を見つける
  6. 70点以上の人数を数える
ステップ2：パターン認識
  • 「すべての点数に対して〇〇する」という繰り返しパターン
  • 「最大値・最小値を求める」というパターン
ステップ3：抽象化
  • 詳細な点数よりも「合格/不合格」という区分が重要
  • 個々の生徒名より「統計データ」が重要
ステップ4：アルゴリズム設計
# 成績データ（本来は入力や読込）
scores = [85, 72, 90, 68, 78, 82, 95, 70, 88, 77,
          92, 65, 80, 75, 88, 70, 73, 85, 79, 82,
          91, 68, 76, 84, 72, 88, 80, 75, 87, 78]

# 合計点と平均点
total = sum(scores)
average = total / len(scores)

# 最高点と最低点
highest = max(scores)
lowest = min(scores)

# 合格者数（70点以上）
passed = 0
for score in scores:
    if score >= 70:
        passed += 1

# 結果表示
print(f"受験者数：{len(scores)}名")
print(f"平均点：{average:.1f}点")
print(f"最高点：{highest}点")
print(f"最低点：{lowest}点")
print(f"合格者数：{passed}名（{passed/len(scores)*100:.1f}%）")

第4部：デバッグの基本（30分）

4.1 エラーの種類

1. 構文エラー（Syntax Error）：文法の間違い
# NG例：コロンがない
if score >= 80
    print("合格")

# エラーメッセージ
# SyntaxError: invalid syntax

# 修正後
if score >= 80:
    print("合格")
2. 実行時エラー（Runtime Error）：実行中に発生
# NG例：0で割り算
result = 100 / 0

# エラーメッセージ
# ZeroDivisionError: division by zero

# 修正：事前にチェック
divisor = 0
if divisor != 0:
    result = 100 / divisor
else:
    print("0では割れません")
3. 論理エラー（Logic Error）：動くが結果が間違い
# NG例：平均点の計算ミス
scores = [80, 90, 70]
average = sum(scores) / 2  # 間違い：2で割っている

print(f"平均点：{average}")  # 120.0（明らかにおかしい）

# 修正
average = sum(scores) / len(scores)  # 正しい
print(f"平均点：{average}")  # 80.0

1. print文デバッグ

        print(f"{students[i]}: {scores[i]}点")

3. 段階的実行
# 複雑な処理を分解して確認
def process_data(data):
    # ステップ1：データ確認
    print(f"入力データ：{data}")
    
    # ステップ2：処理前確認
    filtered = [x for x in data if x >= 70]
    print(f"フィルタ後：{filtered}")
    
    # ステップ3：最終結果
    average = sum(filtered) / len(filtered)
    print(f"平均：{average}")
    
    return average

scores = [85, 65, 90, 72, 58, 88]
result = process_data(scores)

仕様
  • 5人分の生徒名とテスト点数を管理
  • 各生徒の合格/不合格判定（70点以上で合格）
  • クラス全体の平均点を計算
  • 最高点と最低点の生徒を表示
実装例
# データ準備
students = {
    "太郎": 85,
    "花子": 92,
    "次郎": 68,
    "桃子": 78,
    "五郎": 88
}

print("=" * 40)
print("成績処理システム")
print("=" * 40)

# 個別判定
passed_count = 0
for name, score in students.items():
    if score >= 70:
        result = "合格"
        passed_count += 1
    else:
        result = "不合格"
    print(f"{name}：{score}点 → {result}")

print("-" * 40)

# クラス統計
scores = list(students.values())
average = sum(scores) / len(scores)
highest_score = max(scores)
lowest_score = min(scores)

# 最高点・最低点の生徒名を取得
for name, score in students.items():
    if score == highest_score:
        top_student = name
    if score == lowest_score:
        bottom_student = name

print(f"クラス平均：{average:.1f}点")
print(f"最高点：{top_student}（{highest_score}点）")
print(f"最低点：{bottom_student}（{lowest_score}点）")
print(f"合格者：{passed_count}名 / {len(students)}名")
print("=" * 40)

5.3 課題3：教材用ミニゲーム「数当てゲーム」

仕様
  • コンピュータが1~100のランダムな数字を決める
  • プレイヤーが数字を推測
  • 「もっと大きい」「もっと小さい」とヒントを出す
  • 正解したら試行回数を表示
実装例（抜粋）
import random

print("=" * 50)
print("数当てゲーム")
print("=" * 50)
print("1~100の数字を当ててください")
print("-" * 50)

# 正解の数字をランダムに決定
answer = random.randint(1, 100)
attempts = 0
max_attempts = 10

while attempts < max_attempts:
    try:
        guess = int(input(f"予想（残り{max_attempts - attempts}回）："))
        attempts += 1
        
        if guess < 1 or guess > 100:
            print("1~100の範囲で入力してください")
            attempts -= 1
            continue
        
        if guess == answer:
            print("-" * 50)
            print(f"🎉 正解！ 答えは{answer}でした！")
            print(f"試行回数：{attempts}回")
            print("=" * 50)
            break
        elif guess < answer:
            print("⬆ もっと大きい数字です")
        else:
            print("⬇ もっと小さい数字です")
    
    except ValueError:
        print("数字で入力してください")
        attempts -= 1

else:
    print("-" * 50)
    print(f"😢 ゲームオーバー！ 答えは{answer}でした")
    print("=" * 50)

💡 指導のポイント（ICT支援員向け）

児童生徒のつまずきポイントと対応

1. インデントの理解
# よくあるミス
if score >= 80:
print("合格")  # インデントがない

# 正しい形
if score >= 80:
    print("合格")  # 4スペース字下げ

対応：「Pythonではスペースがとても重要」と強調し、視覚的に示す

2. 変数名の命名
# 悪い例
a = 85
b = 90
c = a + b

# 良い例
math_score = 85
english_score = 90
total_score = math_score + english_score

対応：「変数名は内容がわかる名前にする」習慣づけ

3. 比較演算子と代入演算子の混同
# NG
if score = 80:  # 代入演算子（=）を使っている

# OK
if score == 80:  # 比較演算子（==）

対応：「=は代入、==は比較」と明確に区別

4. リストのインデックス
students = ["太郎", "花子", "次郎"]
# インデックスは0から始まる
print(students[0])  # 太郎
print(students[3])  # エラー！（範囲外）

対応：「コンピュータは0から数える」ことを繰り返し説明
